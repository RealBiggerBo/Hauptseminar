	\subsection{Mathematische Definition}
	\begin{frame}
		\begin{iodefinition}
			\begin{itemize}
				\item Signatur $sig(A)$	
				\item Menge an Zuständen $states(A)$
				\item nichtleere Menge an Startzuständen $start(A) \subseteq states(A)$
				\item Übergangsrelation
				\[
				trans(A) \subseteq states(A) \times acts(sig(A)) \times states(A)
				\]
				\item Aufgabenpartition $tasks(A)$
			\end{itemize}
		\end{iodefinition}
	\end{frame}
	
	\newpage
	\subsection{Ausführung}
	\begin{frame}
		\begin{execdefinition}
			\begin{itemize}
				\item Ausführungsfragment $s_0, \pi_1, s_1, \pi_2,...$
				\item $\forall i \geq 0. (s_i, \pi_{i+1}, s_{i+1}) \in trans(A)$
				\item Ausführung falls $s_0 \in starts(A)$
			\end{itemize}
		\end{execdefinition}
	\end{frame}
